 167  {
 168      global $db, $pforumcache, $fid, $forum, $lang;
 169      if(!is_array($pforumcache))

 173          while($forumnav = $db->fetch_array($query))

 175              $pforumcache[$forumnav['pid']][$forumnav['fid']] = $forumnav;
 176          }
 177          unset($forumnav);
 178      }
 179      if(is_array($pforumcache[$pid]))

 188              }
