 582  function get_plugins_list()
 583  {
 584      // Get a list of the plugin files which exist in the plugins directory
 585      $dir = @opendir(MYBB_ROOT."inc/plugins/");
 586      if($dir)
 587      {
 588          while($file = readdir($dir))
 589          {
 590              $ext = get_extension($file);
 591              if($ext == "php")
 592              {
 593                  $plugins_list[] = $file;
 594              }
 595          }
 596          @sort($plugins_list);
 597      }
 598      @closedir($dir);
 599      
 600      return $plugins_list;
 601  }
