 197      /**
 198       * Generate a breadcrumb trail.
 199       */
 200  	function _generate_breadcrumb()
 201      {
 202          if(!is_array($this->_breadcrumb_trail))
 203          {
 204              return false;
 205          }
 206          $trail = "";
 207          foreach($this->_breadcrumb_trail as $key => $crumb)
 208          {
 209              if($this->_breadcrumb_trail[$key+1])
 210              {
 211                  $trail .= "<a href=\"".$crumb['url']."\">".$crumb['name']."</a>";
 212                  if($this->_breadcrumb_trail[$key+2])
 213                  {
 214                      $trail .= " &raquo; ";
 215                  }
 216              }
 217              else
 218              {
 219                  $trail .= "<span class=\"active\">".$crumb['name']."</span>";
 220              }
 221          }
 222          return $trail;
 223      }

 540      /**
 541       * Build the actual navigation menu.
 542       */
 543  	function _build_menu()
 544      {
 545          if(!is_array($this->_menu))
 546          {
 547              return false;
 548          }
 549          $build_menu = "<div id=\"menu\">\n<ul>\n";
 550          ksort($this->_menu);
 551          foreach($this->_menu as $items)
 552          {
 553              foreach($items as $menu_item)
 554              {
 555                  $menu_item['link'] = htmlspecialchars($menu_item['link']);
 556                  if($menu_item['id'] == $this->active_module)
 557                  {
 558                      $sub_menu = $menu_item['submenu'];
 559                      $sub_menu_title = $menu_item['title'];
 560                      $build_menu .= "<li><a href=\"{$menu_item['link']}\" class=\"active\">{$menu_item['title']}</a></li>\n";
 561  
 562                  }
 563                  else
 564                  {
 565                      $build_menu .= "<li><a href=\"{$menu_item['link']}\">{$menu_item['title']}</a></li>\n";
 566                  }
 567              }
 568          }
 569          $build_menu .= "</ul>\n</div>";
 570  
 571          if($sub_menu)
 572          {
 573              $this->_build_submenu($sub_menu_title, $sub_menu);
 574          }
 575          return $build_menu;
 576      }
