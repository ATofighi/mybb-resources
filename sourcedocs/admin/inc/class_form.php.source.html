 336  	function generate_select_box($name, $option_list, $selected=array(), $options=array())
 337      {
 338          if(!isset($options['multiple']))
 339          {
 340              $select = "<select name=\"{$name}\"";
 341          }
 342          else
 343          {
 344              $select = "<select name=\"{$name}\" multiple=\"multiple\"";
 345              if(!isset($options['size']))
 346              {
 347                  $options['size'] = count($option_list);
 348              }
 349          }
 350          if(isset($options['class']))
 351          {
 352              $select .= " class=\"{$options['class']}\"";
 353          }
 354          if(isset($options['id']))
 355          {
 356              $select .= " id=\"{$options['id']}\"";
 357          }
 358          if(isset($options['size']))
 359          {
 360              $select .= " size=\"{$options['size']}\"";
 361          }
 362          $select .= ">\n";
 363          foreach($option_list as $value => $option)
 364          {
 365              $select_add = '';
 366              if(!empty($selected) && ((string)$value == (string)$selected || (is_array($selected) && in_array((string)$value, $selected))))
 367              {
 368                  $select_add = " selected=\"selected\"";
 369              }
 370              $select .= "<option value=\"{$value}\"{$select_add}>{$option}</option>\n";
 371          }
 372          $select .= "</select>\n";
 373          return $select;
 374      }
