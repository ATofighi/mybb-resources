 632      {
 633          return;
 634      }
 635      
 636      $values = explode(";", $values);
 637      foreach($values as $value)
 638      {
 639          $value = trim($value);

 641          list($property, $css_value) = explode(":", $value, 2);
 642          $property = trim($property);

 644          {

 646              case "color":
 647              case "width":

 650              case "font-weight":

 704          $css = preg_replace("#\s{1,}\{#", " {", $css);
 705          $existing_block = $parsed_css[$selector];
 706          
 707          $break = strrpos($selector, "_");
 708          if($break !== false)
 709          {
 710              $actual_occurance = intval(substr($selector, ($break+1)));
 711          }
 712          
 713          if(!$actual_occurance)
 714          {
 715              $actual_occurance = 1;
 716          }
 717          
 718          $occurance = 1;
 719          $pos = 0;
 720          do
 721          {
 722              $pos = strpos($css, "\n".$existing_block['class_name']." {", $pos);
 723              if($pos === false)
 724              {
 725                  break;
 726              }
 727              if($occurance == $actual_occurance)
 728              {
 729                  // This is the part we want to replace, now we need to fetch the opening & closing braces
 730                  $opening = strpos($css, "{", $pos);
 731                  $closing = strpos($css, "}", $pos);
 732                  $css = substr_replace($css, "\n".$generated_css."\n", $opening+1, $closing-$opening-1);
 733                  break;
 734              }
 735              ++$occurance;
 736              ++$pos;
 737          } while($occurance <= $actual_occurance);
 738      }
 739      $css = preg_replace("#{\n*#s", "{\n", $css);
 740      $css = preg_replace("#\s*\}\s*#", "\n}\n\n", $css);
 741      return $css;    
 742  }

 761  function update_theme_stylesheet_list($tid)
 762  {
 763      global $db;
 764      
 765      $stylesheets = array();
 766      
 767      $child_list = make_child_theme_list($tid);
 768      $parent_list = make_parent_theme_list($tid);
 769      
 770      if(!is_array($parent_list))
 771      {
 772          return false;
 773      }

 775      $tid_list = implode(',', $parent_list);
 776      
 777      // Get our list of stylesheets
 778      $query = $db->simple_select("themestylesheets", "sid,cachefile,attachedto,tid,lastmodified,name", "tid IN ({$tid_list})", array('order_by' => 'tid', 'order_dir' => 'desc'));
 779      while($stylesheet = $db->fetch_array($query))
 780      {
 781          if(!$stylesheets[$stylesheet['name']])
 782          {
 783              if($stylesheet['tid'] != $tid)
 784              {
 785                  $stylesheet['inherited'] = $stylesheet['tid'];
 786              }
 787              
 788              $stylesheets[$stylesheet['name']] = $stylesheet;
 789          }
 790      }
 791      
 792      foreach($stylesheets as $name => $stylesheet)
 793      {
 794          $sid = $stylesheet['sid'];
 795          $css_url = "css.php?stylesheet={$sid}";

 799              if(file_exists(MYBB_ROOT."cache/themes/theme{$theme_id}/{$stylesheet['name']}") && filemtime(MYBB_ROOT."cache/themes/theme{$theme_id}/{$stylesheet['name']}") >= $stylesheet['lastmodified'])
 800              {
 801                  $css_url = "cache/themes/theme{$theme_id}/{$stylesheet['name']}";
 802                  break;
 803              }
 804          }
 805          
 806          $attachedto = $stylesheet['attachedto'];
 807          if(!$attachedto)
 808          {
 809              $attachedto = "global";
 810          }
 811          // private.php?compose,folders|usercp.php,global|global
 812          $attachedto = explode("|", $attachedto);
 813          foreach($attachedto as $attached_file)
 814          {
 815              $attached_actions = array();
 816              if(strpos($attached_file, '?') !== false)
 817              {
 818                  $attached_file = explode('?', $attached_file);
 819                  $attached_actions = explode(",", $attached_file[1]);
 820                  $attached_file = $attached_file[0];
 821              }
 822              
 823              if(count($attached_actions) == 0)
 824              {
 825                  $attached_actions = array("global");
 826              }
 827              
 828              foreach($attached_actions as $action)
 829              {
 830                  $theme_stylesheets[$attached_file][$action][] = $css_url;
 831                  
 832                  if($stylesheet['inherited'])
 833                  {
 834                      $theme_stylesheets['inherited']["{$attached_file}_{$action}"][$css_url] = $stylesheet['inherited'];
 835                  }
 836              }
 837          }
 838      }
 839  
 840      // Now we have our list of built stylesheets, save them
 841      $updated_theme = array(
 842          "stylesheets" => $db->escape_string(serialize($theme_stylesheets))
 843      );
 844      $db->update_query("themes", $updated_theme, "tid='{$tid}'");
 845      
 846      // Do we have any children themes that need updating too?
 847      if(count($child_list) > 0)
 848      {
 849          foreach($child_list as $id)
 850          {
 851              update_theme_stylesheet_list($id);
 852          }
 853      }
 854      
 855      return true;
 856  }
