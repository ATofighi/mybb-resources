 308  function js_array_to_php_array(a)
 309  {
 310      var a_php = "";
 311      var total = 0;
 312      for(var key in a)
 313      {
 314          ++total;
 315          a_php = a_php + "s:" +
 316                  String(key).length + ":\"" + String(key) + "\";s:" +
 317                  String(a[key]).length + ":\"" + String(a[key]) + "\";";
 318      }
 319      a_php = "a:" + total + ":{" + a_php + "}";
 320      return a_php;
 321  }
