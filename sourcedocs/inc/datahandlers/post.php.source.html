 107  	function verify_author()
 108      {
 109          global $mybb;
 110  
 111          $post = &$this->data;
 112  
 113          // Don't have a user ID at all - not good (note, a user id of 0 will still work).
 114          if(!isset($post['uid']))
 115          {
 116              $this->set_error("invalid_user_id");
 117              return false;
 118          }
 119          // If we have a user id but no username then fetch the username.
 120          else if($post['uid'] > 0 && $post['username'] == '')
 121          {
 122              $user = get_user($post['uid']);
 123              $post['username'] = $user['username'];
 124          }
 125  
 126          // After all of this, if we still don't have a username, force the username as "Guest" (Note, this is not translatable as it is always a fallback)
 127          if(!$post['username'])
 128          {
 129              $post['username'] = "Guest";
 130          }
 131  
 132          // Sanitize the username
 133          $post['username'] = htmlspecialchars_uni($post['username']);
 134          return true;
 135      }

 143  	function verify_subject()
 144      {
 145          global $db;
 146          $post = &$this->data;
 147          $subject = &$post['subject'];
 148          $subject = trim_blank_chrs($subject);
 149  
 150          // Are we editing an existing thread or post?
 151          if($this->method == "update" && $post['pid'])
 152          {
 153              if(!$post['tid'])
 154              {
 155                  $query = $db->simple_select("posts", "tid", "pid='".intval($post['pid'])."'");
 156                  $post['tid'] = $db->fetch_field($query, "tid");
 157              }
 158              // Here we determine if we're editing the first post of a thread or not.
 159              $options = array(
 160                  "limit" => 1,
 161                  "limit_start" => 0,
 162                  "order_by" => "dateline",
 163                  "order_dir" => "asc"
 164              );
 165              $query = $db->simple_select("posts", "pid", "tid='".$post['tid']."'", $options);
 166              $first_check = $db->fetch_array($query);
 167              if($first_check['pid'] == $post['pid'])
 168              {
 169                  $first_post = true;
 170              }
 171              else
 172              {
 173                  $first_post = false;
 174              }
 175  
 176              // If this is the first post there needs to be a subject, else make it the default one.
 177              if(my_strlen($subject) == 0 && $first_post)
 178              {
 179                  $this->set_error("firstpost_no_subject");
 180                  return false;
 181              }
 182              elseif(my_strlen($subject) == 0)
 183              {
 184                  $thread = get_thread($post['tid']);
 185                  $subject = "RE: ".$thread['subject'];
 186              }
 187          }
 188  
 189          // This is a new post
 190          else if($this->action == "post")
 191          {
 192              if(my_strlen($subject) == 0)
 193              {
 194                  $thread = get_thread($post['tid']);
 195                  $subject = "RE: ".$thread['subject'];
 196              }
 197          }
 198  
 199          // This is a new thread and we require that a subject is present.
 200          else
 201          {
 202              if(my_strlen($subject) == 0)
 203              {
 204                  $this->set_error("missing_subject");
 205                  return false;
 206              }
 207          }
 208  
 209          if(my_strlen($subject) > 85)
 210          {
 211              // Subject is too long
 212              $this->set_error('subject_too_long', my_strlen($subject));
 213              return false;
 214          }
 215  
 216          // Subject is valid - return true.
 217          return true;

 278          global $mybb;
 279  
 280          $post = &$this->data;

 284          {
 285              if($this->verify_post_merge(true) !== true)
 286              {
 287                  return true;

 291              $user = get_user($post['uid']);

 294              if(TIME_NOW-$user['lastpost'] <= $mybb->settings['postfloodsecs'] && !is_moderator($post['fid'], "", $user['uid']))
 295              {
 296                  // Oops, user has been flooding - throw back error message.
 297                  $time_to_wait = ($mybb->settings['postfloodsecs'] - (TIME_NOW-$user['lastpost'])) + 1;
 298                  if($time_to_wait == 1)
 299                  {
 300                      $this->set_error("post_flooding_one_second");
 301                  }
 302                  else
 303                  {
 304                      $this->set_error("post_flooding", array($time_to_wait));
 305                  }
 306                  return false;
 307              }
 308          }
 309          // All is well that ends well - return true.
 310          return true;

 313  	function verify_post_merge($simple_mode=false)
 314      {
 315          global $mybb, $db, $session;
 316          
 317          $post = &$this->data;
 318          
 319          // Are we starting a new thread?
 320          if(!$post['tid'])
 321          {
 322              return true;
 323          }
 324          
 325          // Are we even turned on?
 326          if(empty($mybb->settings['postmergemins']))
 327          {
 328              return true;
 329          }
 330          
 331          // Assign a default separator if none is specified
 332          if(trim($mybb->settings['postmergesep']) == "")
 333          {
 334              $mybb->settings['postmergesep'] = "[hr]";
 335          }
 336          
 337          // Check to see if this person is in a usergroup that is excluded
 338          if(trim($mybb->settings['postmergeuignore']) != "")
 339          {
 340              $gids = explode(',', $mybb->settings['postmergeuignore']);
 341              $gids = array_map('intval', $gids);
 342              
 343              
 344              $user_usergroups = explode(',', $mybb->user['usergroup'].",".$mybb->user['additionalgroups']);
 345              if(count(array_intersect($user_usergroups, $gids)) > 0)
 346              {
 347                  return true;
 348              }            
 349          }
 350          
 351          // Select the lastpost and fid information for this thread
 352          $query = $db->simple_select("threads", "lastpost,fid", "lastposteruid='".$post['uid']."' AND tid='".$post['tid']."'", array('limit' => '1'));
 353          $thread = $db->fetch_array($query);
 354          
 355          // Check to see if the same author has posted within the merge post time limit
 356          if((intval($mybb->settings['postmergemins']) != 0 && trim($mybb->settings['postmergemins']) != "") && (TIME_NOW-$thread['lastpost']) > (intval($mybb->settings['postmergemins'])*60))
 357          {
 358              return true;
 359          }
 360          
 361          if(strstr($mybb->settings['postmergefignore'], ','))
 362          {
 363              $fids = explode(',', $mybb->settings['postmergefignore']);
 364              foreach($fids as $key => $forumid)
 365              {
 366                  $fid[] = intval($forumid);
 367              }
 368              
 369              if(in_array($thread['fid'], $fid))
 370              {
 371                  return true;
 372              }
 373              
 374          }
 375          else if(trim($mybb->settings['postmergefignore']) != "" && $thread['fid'] == intval($mybb->settings['postmergefignore']))
 376          {
 377              return true;
 378          }
 379          
 380          if($simple_mode == true)
 381          {
 382              return false;
 383          }
 384          
 385          if($post['uid'])
 386          {
 387              $user_check = "uid='".$post['uid']."'";
 388          }
 389          else
 390          {
 391              $user_check = "ipaddress='".$db->escape_string($session->ipaddress)."'";
 392          }
 393          
 394          $query = $db->simple_select("posts", "pid,message,visible,posthash", "{$user_check} AND tid='".$post['tid']."' AND dateline='".$thread['lastpost']."'", array('order_by' => 'pid', 'order_dir' => 'DESC', 'limit' => 1));
 395          return $db->fetch_array($query);
 396      }

 419  
 420          // Check if this post contains more images than the forum allows

 422          {
 423              require_once  MYBB_ROOT."inc/class_parser.php";
 424              $parser = new postParser;
 425  
 426              // Parse the message.
 427              $parser_options = array(
 428                  "allow_html" => $forum['allowhtml'],
 429                  "allow_mycode" => $forum['allowmycode'],
 430                  "allow_imgcode" => $forum['allowimgcode'],
 431                  "allow_videocode" => $forum['allowvideocode'],
 432                  "filter_badwords" => 1
 433              );
 434  
 435              if($post['options']['disablesmilies'] != 1)
 436              {
 437                  $parser_options['allow_smilies'] = $forum['allowsmilies'];
 438              }
 439              else
 440              {
 441                  $parser_options['allow_smilies'] = 0;
 442              }
 443  
 444              $image_check = $parser->parse_message($post['message'], $parser_options);
 445  
 446              // And count the number of image tags in the message.
 447              $image_count = substr_count($image_check, "<img");
 448              if($image_count > $mybb->settings['maxpostimages'])
 449              {
 450                  // Throw back a message if over the count with the number of images as well as the maximum number of images per post.
 451                  $this->set_error("too_many_images", array(1 => $image_count, 2 => $mybb->settings['maxpostimages']));

 661  
 662          $post = &$this->data;
 663  
 664          // Yes, validating is required.
 665          if(!$this->get_validated())
 666          {
 667              die("The post needs to be validated before inserting it into the DB.");
 668          }
 669          if(count($this->get_errors()) > 0)
 670          {
 671              die("The post is not valid.");
 672          }
 673          
 674          // Fetch the thread
 675          $thread = get_thread($post['tid']);
 676  
 677          // This post is being saved as a draft.
 678          if($post['savedraft'])
 679          {
 680              $visible = -2;
 681          }
 682          
 683          // Otherwise this post is being made now and we have a bit to do.
 684          else
 685          {
 686              // Automatic subscription to the thread
 687              if($post['options']['subscriptionmethod'] != "" && $post['uid'] > 0)
 688              {
 689                  switch($post['options']['subscriptionmethod'])
 690                  {
 691                      case "instant":
 692                          $notification = 1;
 693                          break;
 694                      default:
 695                          $notification = 0;
 696                  }
 697  
 698                  require_once  MYBB_ROOT."inc/functions_user.php";
 699                  add_subscribed_thread($post['tid'], $notification, $post['uid']);
 700              }
 701  
 702              // Perform any selected moderation tools.
 703              if(is_moderator($post['fid'], "", $post['uid']))
 704              {
 705                  $lang->load($this->language_file, true);
 706  
 707                  $modoptions = $post['modoptions'];
 708                  $modlogdata['fid'] = $thread['fid'];
 709                  $modlogdata['tid'] = $thread['tid'];
 710  
 711                  // Close the thread.
 712                  if($modoptions['closethread'] == 1 && $thread['closed'] != 1)
 713                  {
 714                      $newclosed = "closed=1";
 715                      log_moderator_action($modlogdata, $lang->thread_closed);
 716                  }
 717  
 718                  // Open the thread.
 719                  if($modoptions['closethread'] != 1 && $thread['closed'] == 1)
 720                  {
 721                      $newclosed = "closed=0";
 722                      log_moderator_action($modlogdata, $lang->thread_opened);
 723                  }
 724  
 725                  // Stick the thread.
 726                  if($modoptions['stickthread'] == 1 && $thread['sticky'] != 1)
 727                  {
 728                      $newstick = "sticky='1'";
 729                      log_moderator_action($modlogdata, $lang->thread_stuck);
 730                  }
 731  
 732                  // Unstick the thread.
 733                  if($modoptions['stickthread'] != 1 && $thread['sticky'])
 734                  {
 735                      $newstick = "sticky='0'";                    
 736                      log_moderator_action($modlogdata, $lang->thread_unstuck);
 737                  }
 738  
 739                  // Execute moderation options.
 740                  if($newstick && $newclosed)
 741                  {
 742                      $sep = ",";
 743                  }
 744                  if($newstick || $newclosed)
 745                  {
 746                      $db->write_query("
