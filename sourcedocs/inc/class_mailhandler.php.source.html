 109  	function build_message($to, $subject, $message, $from="", $charset="", $headers="", $format="text", $message_text="", $return_email="")
 110      {

 112

 294          if(isset($_SERVER['SERVER_NAME']))
 295          {
 296              $http_host = $_SERVER['SERVER_NAME'];
 297          }
 298          else if(isset($_SERVER['HTTP_HOST']))
 299          {
 300              $http_host = $_SERVER['HTTP_HOST'];
 301          }
 302          else
 303          {
 304              $http_host = "unknown.local";
 305          }
 306  
 307          $msg_id = md5(uniqid(TIME_NOW)) . "@" . $http_host;
 308  
 309          if($mybb->settings['mail_message_id'])
 310          {
 311              $this->headers .= "Message-ID: <{$msg_id}>{$this->delimiter}";
 312          }
 313          $this->headers .= "Content-Transfer-Encoding: 8bit{$this->delimiter}";
 314          $this->headers .= "X-Priority: 3{$this->delimiter}";
 315          $this->headers .= "X-MSMail-Priority: Normal{$this->delimiter}";
 316          $this->headers .= "X-Mailer: MyBB{$this->delimiter}";
 317          $this->headers .= "MIME-Version: 1.0{$this->delimiter}";

 374      /**
 375       * Encode a string based on the character set enabled. Used to encode subjects
 376       * and recipients in email messages going out so that they show up correctly
 377       * in email clients.
 378       *
 379       * @param string The string to be encoded.
 380       * @return string The encoded string.
 381       */
 382  	function utf8_encode($string)
 383      {
 384          if(strtolower($this->charset) == 'utf-8' && preg_match('/[^\x20-\x7E]/', $string))
 385          {
 386              $chunk_size = 47; // Derived from floor((75 - strlen("=?UTF-8?B??=")) * 0.75);
 387              $len = strlen($string);
 388              $output = '';
 389              $pos = 0;
 390  
 391              while($pos < $len)
 392              {
 393                  $newpos = min($pos + $chunk_size, $len);
 394  
 395                  while(ord($string[$newpos]) >= 0x80 && ord($string[$newpos]) < 0xC0)
 396                  {
 397                      // Reduce len until it's safe to split UTF-8.
 398                      $newpos--;
 399                  }
 400  
 401                  $chunk = substr($string, $pos, $newpos - $pos);
 402                  $pos = $newpos;
 403  
 404                  $output .= " =?UTF-8?B?".base64_encode($chunk)."?=\n";
 405              }
 406              return trim($output);
 407          }
 408          return $string;
 409      }
