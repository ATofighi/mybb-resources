 292  * parser_php_info
 293  * Function to get and parse the list of PHP info into a usuable array
 294  *
 295  * @return Array An array of all the extensions installed in PHP
 296  */
 297  function parse_php_info()
 298  {
 299      ob_start();
 300      phpinfo(INFO_MODULES);
 301      $phpinfo_html = ob_get_contents();
 302      ob_end_clean();
 303  
 304      $phpinfo_html = strip_tags($phpinfo_html, "<h2><th><td>");
 305      $phpinfo_html = preg_replace("#<th[^>]*>([^<]+)<\/th>#", "<info>$1</info>", $phpinfo_html);
 306      $phpinfo_html = preg_replace("#<td[^>]*>([^<]+)<\/td>#", "<info>$1</info>", $phpinfo_html);
 307      $phpinfo_html = preg_split("#(<h2[^>]*>[^<]+<\/h2>)#", $phpinfo_html, -1, PREG_SPLIT_DELIM_CAPTURE);
 308      $modules = array();
 309      
 310      for($i=1; $i < count($phpinfo_html); $i++)
 311      {
 312          if(preg_match("#<h2[^>]*>([^<]+)<\/h2>#", $phpinfo_html[$i], $match))
 313          {
 314              $name = trim($match[1]);
 315              $tmp2 = explode("\n", $phpinfo_html[$i+1]);
 316              foreach($tmp2 as $one)
 317              {
 318                  $pat = '<info>([^<]+)<\/info>';
 319                  $pat3 = "/$pat\s*$pat\s*$pat/";
 320                  $pat2 = "/$pat\s*$pat/";
 321                  
 322                  // 3 columns
 323                  if(preg_match($pat3, $one, $match))
 324                  {
 325                      $modules[$name][trim($match[1])] = array(trim($match[2]), trim($match[3]));
 326                  }
 327                  // 2 columns
 328                  else if(preg_match($pat2, $one, $match))
 329                  {
 330                      $modules[$name][trim($match[1])] = trim($match[2]);
 331                  }
 332              }
 333          }
 334      }
 335      return $modules;
 336  }
