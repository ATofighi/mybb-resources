 152          // Ignore potentially missing plugins.
 153          if(!file_exists(MYBB_ROOT."inc/plugins/".$plugin.".php"))
 154          {
 155              return true;
 156          }
 157  
 158          require_once MYBB_ROOT."inc/plugins/".$plugin.".php";
 159  
 160          $info_func = "{$plugin}_info";
 161          if(!function_exists($info_func))
 162          {
 163              return false;
 164          }
 165          $plugin_info = $info_func();
 166          
 167          // No compatibility set or compatibility = * - assume compatible
 168          if(!$plugin_info['compatibility'] || $plugin_info['compatibility'] == "*")
 169          {
 170              return true;
 171          }
 172          $compatibility = explode(",", $plugin_info['compatibility']);
 173          foreach($compatibility as $version)
 174          {
 175              $version = trim($version);
 176              $version = str_replace("*", ".+", preg_quote($version));
 177              $version = str_replace("\.+", ".+", $version);
 178              if(preg_match("#{$version}#i", $mybb->version_code))
 179              {
 180                  return true;
 181              }
 182          }
 183  
 184          // Nothing matches
 185          return false;
