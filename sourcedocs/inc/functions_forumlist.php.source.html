 416  /**
 417   * Fetch the status indicator for a forum based on its last post and the read date
 418   *
 419   * @param array Array of information about the forum
 420   * @param array Array of information about the lastpost date
 421   * @return array Array of the folder image to be shown and the alt text
 422   */
 423  function get_forum_lightbulb($forum, $lastpost, $locked=0)
 424  {
 425      global $mybb, $lang, $db, $unread_forums;
 426  
 427      // This forum is closed, so override the folder icon with the "offlock" icon.
 428      if($forum['open'] == 0 || $locked)
 429      {
 430          $folder = "offlock";
 431          $altonoff = $lang->forum_locked;
 432      }
 433      else
 434      {
 435          // Fetch the last read date for this forum
 436          if($forum['lastread'])
 437          {
 438              $forum_read = $forum['lastread'];
 439          }
 440          elseif($mybb->cookies['mybb']['readallforums'])
 441          {
 442              // We've hit the read all forums as a guest, so use the lastvisit of the user
 443              $forum_read = $mybb->cookies['mybb']['lastvisit'];
 444          }
 445          else
 446          {
 447              $forum_read = 0;
 448              $threadcut = TIME_NOW - 60*60*24*$mybb->settings['threadreadcut'];
 449  
 450              // If the user is a guest, do they have a forumsread cookie?
 451              if(!$mybb->user['uid'] && $mybb->cookies['mybb']['forumread'])
 452              {
 453                  // If they've visited us before, then they'll have this cookie - otherwise everything is unread...
 454                  $forum_read = my_get_array_cookie("forumread", $forum['fid']);
 455              }
 456              else if($mybb->user['uid'] && $mybb->settings['threadreadcut'] > 0 && $threadcut > $lastpost['lastpost'])
 457              {
 458                  // We have a user, the forum's unread and we're over our threadreadcut limit for the lastpost - we mark these as read
 459                  $forum_read = $lastpost['lastpost'] + 1;
 460              }
 461          }
 462  
 463          //if(!$forum_read)
 464          //{
 465              //$forum_read = $mybb->user['lastvisit'];
 466          //}
 467          
 468           // If the lastpost is greater than the last visit and is greater than the forum read date, we have a new post 
 469          if($lastpost['lastpost'] > $forum_read && $lastpost['lastpost'] != 0) 
 470          {
 471              $unread_forums++;
 472              $folder = "on";
 473              $altonoff = $lang->new_posts;
 474          }
 475          // Otherwise, no new posts
 476          else
 477          {
 478              $folder = "off";
 479              $altonoff = $lang->no_new_posts;
 480          }
 481      }
 482  
 483      return array(
 484          "folder" => $folder,
 485          "altonoff" => $altonoff
 486      );
 487  }

 489  /**
 490   * Fetch the number of unapproved posts, formatted, from a forum
 491   *
 492   * @param array Array of information about the forum
 493   * @return array Array containing formatted string for posts and string for threads
 494   */
 495  function get_forum_unapproved($forum)
 496  {
 497      global $lang;
 498  
 499      $unapproved_threads = $unapproved_posts = '';
 500  
 501      // If the user is a moderator we need to fetch the count
 502      if(is_moderator($forum['fid']))
 503      {
 504          // Forum has one or more unaproved posts, format language string accordingly
 505          if($forum['unapprovedposts'])
 506          {
 507              if($forum['unapprovedposts'] > 1)
 508              {
 509                  $unapproved_posts_count = $lang->sprintf($lang->forum_unapproved_posts_count, $forum['unapprovedposts']);
 510              }
 511              else
 512              {
 513                  $unapproved_posts_count = $lang->sprintf($lang->forum_unapproved_post_count, 1);
 514              }
 515              $unapproved_posts = " <span title=\"{$unapproved_posts_count}\">(".my_number_format($forum['unapprovedposts']).")</span>";
 516          }
 517          // Forum has one or more unapproved threads, format language string accordingly
 518          if($forum['unapprovedthreads'])
 519          {
 520              if($forum['unapprovedthreads'] > 1)
 521              {
 522                  $unapproved_threads_count = $lang->sprintf($lang->forum_unapproved_threads_count, $forum['unapprovedthreads']);
 523              }
 524              else
 525              {
 526                  $unapproved_threads_count = $lang->sprintf($lang->forum_unapproved_thread_count, 1);
 527              }
 528              $unapproved_threads = " <span title=\"{$unapproved_threads_count}\">(".my_number_format($forum['unapprovedthreads']).")</span>";
 529          }
 530      }
 531      return array(
 532          "unapproved_posts" => $unapproved_posts,
 533          "unapproved_threads" => $unapproved_threads
 534      );
