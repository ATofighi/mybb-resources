  86  function archive_navigation()
  87  {
  88      global $navbits, $mybb, $lang;
  89  
  90      $navsep = " &gt; ";
  91      if(is_array($navbits))
  92      {
  93          reset($navbits);
  94          foreach($navbits as $key => $navbit)
  95          {
  96              if($navbits[$key+1])
  97              {
  98                  if($navbits[$key+2])
  99                  {
 100                      $sep = $navsep;
 101                  }
 102                  else
 103                  {
 104                      $sep = "";
 105                  }
 106                  $nav .= "<a href=\"".$navbit['url']."\">".$navbit['name']."</a>$sep";
 107              }
 108          }
 109      }
 110      $navsize = count($navbits);
 111      $navbit = $navbits[$navsize-1];
 112      if($nav)
 113      {
 114          $activesep = $navsep;
 115      }
 116      $nav .= $activesep.$navbit['name'];
 117  
 118      return $nav;
 119  }

 220   * Check the password given on a certain forum for validity
 221   *
 222   * @param int The forum ID
 223   * @param boolean The Parent ID

 225  function check_forum_password_archive($fid, $pid=0)
 226  {
 227      global $forum_cache;
 228      
 229      if(!is_array($forum_cache))
 230      {
 231          $forum_cache = cache_forums();
 232          if(!$forum_cache)
 233          {
 234              return false;
 235          }
 236      }
 237  
 238      // Loop through each of parent forums to ensure we have a password for them too
 239      $parents = explode(',', $forum_cache[$fid]['parentlist']);
 240      rsort($parents);
 241      if(!empty($parents))
 242      {
 243          foreach($parents as $parent_id)
 244          {
 245              if($parent_id == $fid || $parent_id == $pid)
 246              {
 247                  continue;
 248              }
 249              
 250              if($forum_cache[$parent_id]['password'] != "")
 251              {
 252                  check_forum_password_archive($parent_id, $fid);
 253              }
 254          }
 255      }
 256      
 257      $password = $forum_cache[$fid]['password'];
 258      if($password)
 259      {
 260          if(!$mybb->cookies['forumpass'][$fid] || ($mybb->cookies['forumpass'][$fid] && md5($mybb->user['uid'].$password) != $mybb->cookies['forumpass'][$fid]))
 261          {
 262              archive_error_no_permission();
 263          }
 264      }
 265  }
