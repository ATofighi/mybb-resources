  73          WHERE fid='$fid' AND visible='1' AND closed    NOT LIKE 'moved|%'
  74      ");

  78      if(!$count['posts'])

  87          WHERE fid='$fid' AND visible='0' AND closed NOT LIKE 'moved|%'
  88      ");

  95          $count['unapprovedthreads'] = 0;

  97
