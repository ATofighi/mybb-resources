 165  function check_thumbnail_memory($width, $height, $type, $bitdepth, $channels)
 166  {
 167      if(!function_exists("memory_get_usage"))
 168      {
 169          return false;
 170      }

 172      $memory_limit = @ini_get("memory_limit");

 174      {
 175          return false;
 176      }
 177  
 178      $limit = preg_match("#^([0-9]+)\s?([kmg])b?$#i", trim(my_strtolower($memory_limit)), $matches);
 179      $memory_limit = 0;
 180      if($matches[1] && $matches[2])
 181      {
 182          switch($matches[2])

 184              case "k":

 201      {
